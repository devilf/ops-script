# ops-script
  init-centos7.sh:
  
         用于centos7系统初始化
         sh init-centos7.sh menus
	
  send_mail.py:
  
        用于zabbix中的邮件发送
        python3 send_mail.py "要发送的邮件地址" "发送的邮件内容" "发送的主题"

  mv.py:
  
        用于将指定目录下的文件进行批量改名
        python mv.py "/data/log/" "([0-9A-Z_]+)[0-9_]{6}.pdf" "_10_001.pdf"
